useCollector

过滤数组数据

import {useCollector} from "cyl-hooks-tools/useCollector"
const list = [
  {
    id:1,
    name:"你好1"
  },
   {
    id:2,
    name:"你好2"
  }
]
export default function Component(){
    const ids = useCollector(list,t=>t.id)
    return <div>
      {ids}
    </div>
}
